\section*{Program 1: Online Bookstore Management}

\subsection*{Query 1}
\textbf{Problem:} Display all customers whose city name starts with 'A'.  

\textbf{SQL Query:}  
\begin{verbatim}
SELECT * FROM Customer WHERE city LIKE 'A%';
\end{verbatim}

\textbf{Output:}

\begin{tabular}{|c|c|c|}
\hline
customer\_id & name & city \\
\hline
3 & Aarav & Ahmedabad \\
\hline
\end{tabular}
